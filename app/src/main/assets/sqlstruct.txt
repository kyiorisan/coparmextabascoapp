CREATE TABLE IF NOT EXISTS EMPRESAS (RFC TEXT NOT NULL,NOMBRE TEXT NOT NULL,GIRO TEXT, LOGO TEXT,WEB TEXT,EMAIL TEXT,VALOK	INTEGER,VALBAD	INTEGER,ISDEDIEZ INTEGER NOT NULL,SERVICIOS	TEXT,PRIMARY KEY(RFC))
CREATE TABLE IF NOT EXISTS SUCURSALES (ID INTEGER NOT NULL,NAME TEXT,ADDRESS TEXT,PHONE TEXT,EMAIL TEXT,LAT REAL,LON REAL,HRATENCION TEXT,EMP_RFC TEXT NOT NULL,PRIMARY KEY(ID),FOREIGN KEY(EMP_RFC) REFERENCES EMPRESAS(RFC))
CREATE TABLE IF NOT EXISTS OFERTAS (ID INTEGER NOT NULL,TEXT TEXT NOT NULL,BANNER TEXT,FINI TEXT,FEND TEXT,WEBSRCS TEXT,EMP_RFC TEXT NOT NULL,PRIMARY KEY(ID),FOREIGN KEY(EMP_RFC) REFERENCES EMPRESA(RFC)
)
